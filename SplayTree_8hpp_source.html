72 std::shared_ptr<splay_tree_node_t<key_t, value_t>> insert_helper(std::shared_ptr<splay_tree_node_t<key_t, value_t>> root, const key_t &key, const value_t &value) {
73 if (!root) return std::make_shared<splay_tree_node_t<key_t, value_t>>(key, value);
74 root = splay(root, key);
75 if (key < root->key) {

77 temp->rchild = root;

91 std::shared_ptr<splay_tree_node_t<key_t, value_t>> remove_helper(std::shared_ptr<splay_tree_node_t<key_t, value_t>> root, const key_t &key) {
92 if (!root) return nullptr;
93 root = splay(root, key);
94 if (root->key != key) return root;
95 auto temp = root;
96 if (!root->lchild) {
97 root = root->rchild;
98 } else if (!root->rchild) {
99 root = root->lchild;
100 } else {
101 root = splay(root->lchild, key);
102 root->rchild = temp->rchild;
103 }
104 temp.reset();
105 return root;

134 std::shared_ptr<splay_tree_node_t<key_t, value_t>> splay(std::shared_ptr<splay_tree_node_t<key_t, value_t>> root, const key_t &key) {
135 if (!root) return nullptr;
136 if (root->key == key) return root;
137 if (key < root->key) {
138 if (!root->lchild) return root;
139 if (key < root->lchild->key) {
140 root->lchild->lchild = splay(root->lchild->lchild, key);
141 root = rrotate(root);
142 } else if (key > root->lchild->key) {
143 root->lchild->rchild = splay(root->lchild->rchild, key);
144 if (root->lchild->rchild) {
145 root->lchild = lrotate(root->lchild);
146 }
147 }
148 return (root->lchild) ? rrotate(root) : root;
149 } else {
150 if (!root->rchild) return root;
151 if (key < root->rchild->key) {
152 root->rchild->lchild = splay(root->rchild->lchild, key);
153 if (root->rchild->lchild) {
154 root->rchild = rrotate(root->rchild);
155 }
156 } else if (key > root->rchild->key) {
157 root->rchild->rchild = splay(root->rchild->rchild, key);
158 root = lrotate(root);
159 }
160 return (root->rchild) ? lrotate(root) : root;
161 }
162 }
